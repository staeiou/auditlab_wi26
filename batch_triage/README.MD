│

├─ scripts/
│  ├─ generate_batch_payloads.py  # reads config/config.yaml; writes batch/<model>/...
│  ├─ run_openai_batch.py         # “main” CLI entrypoint (small): orchestrates submission + polling
│  └─ replay_vllm.py              # (optional) JSONL replay runner for vLLM server
│
├─ src/
│  ├─ __init__.py
│  ├─ config_loader.py            # load/validate YAMLs; merge overrides; default paths

│  ├─ openai_batch/

│  │  ├─ client.py                # wrappers: upload_file, create_batch, get_batch, download_file
│  │  ├─ validator.py             # enforce limits from limits.yaml (50k req, 200MB, one-model-per-file)
│  │  └─ parser.py                # parse OpenAI batch output jsonl -> compact results rows
│  │
│  ├─ state/
│  │  ├─ __init__.py

│  ├─ wandb_logging/
│  │  ├─ __init__.py
│  │  ├─ run_manager.py           # wandb.init config, tags, run metadata
│  │  ├─ artifacts.py             # upload inputs/outputs/state files as artifacts on a schedule
│  │  └─ tables.py                # optional: wandb.Table logging for small runs
│  │
│  └─ io_utils/
│     ├─ jsonl.py                 # streaming jsonl reader/writer
│     └─ paths.py                 # consistent naming: safe_model_dirname, output paths
│
├─ state/

│     └─ ...
